## 查看当前用户拥有的表

     select table_name from user_tables;

## 查看所有用户的表
  
     select table_name from all_tables;

## 查看当前用户下所有表注释

     select * from user_tab_comments;

## 查看当前用户下某表所有字段注释

     select * from user_col_comments where TABLE_NAME='某表名称'；
